# MG Enterprise — Role Permissions
# ============================================================
# Role permissions are hardcoded in the frontend (src/hooks/usePermissions.ts).
#
# Dev users (is_developer=true) automatically get full access.
#
# Non-dev role defaults:
#   SiteSuperAdmin  → screens, content, production, trucks, team
#   Manager          → screens, content, production, trucks
#   User             → view (Dashboard only)
#
# To change these mappings, edit usePermissions.ts and rebuild.
# ============================================================
